#version 330 core
layout (location = 0) in vec2 aPos;
layout (location = 1) in vec2 texCoords;
layout (location =  2) in vec4 aColor;

out vec2 TexCoords;
out vec4 Color;

uniform mat4 model;
uniform mat4 projection;

void main()
{
    Color = aColor;
    TexCoords = texCoords;
    gl_Position = projection * model * vec4(aPos.x, aPos.y, 0.0, 1.0);
}
